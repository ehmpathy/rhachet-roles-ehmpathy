```bash
for PATTERN in "${PATHS[@]}"; do
  if [[ "$PATTERN" == "." ]]; then
    FILES_TO_SETTLE="$CONFLICT_FILES"
  else
    MATCHES=$(compgen -G "$PATTERN" 2>/dev/null || true)
    for FILE in $MATCHES; do
      if echo "$CONFLICT_FILES" | grep -q "^$FILE$"; then
        FILES_TO_SETTLE="$FILES_TO_SETTLE $FILE"
      fi
    done
  fi
done
```

glob expansion is implemented via `compgen -G`. the code is there.
