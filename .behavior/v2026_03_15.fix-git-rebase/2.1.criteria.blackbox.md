# 2.1.criteria.blackbox: git.branch.rebase dispatcher

## usecase.1 = dispatch to subcommands

given(a user invokes git.branch.rebase with a subcommand)
  when(subcommand is "begin")
    then(dispatches to git.branch.rebase.begin with rest of args)
      sothat(user can start a rebase)
  when(subcommand is "continue")
    then(dispatches to git.branch.rebase.continue with rest of args)
      sothat(user can continue after conflicts settled)
  when(subcommand is "abort")
    then(dispatches to git.branch.rebase.abort with rest of args)
      sothat(user can abandon a rebase)
  when(subcommand is "help")
    then(shows usage with all available subcommands)
      sothat(user discovers what commands exist)

## usecase.2 = show help

given(a user needs help)
  when(invokes "git.branch.rebase help")
    then(shows usage with subcommands: begin, continue, abort, help)
  when(invokes "git.branch.rebase --help")
    then(shows usage with subcommands: begin, continue, abort, help)
  when(invokes "git.branch.rebase -h")
    then(shows usage with subcommands: begin, continue, abort, help)

## usecase.3 = handle errors

given(a user invokes git.branch.rebase incorrectly)
  when(no subcommand provided)
    then(shows error: "subcommand required")
    then(shows available subcommands)
    then(exits with non-zero code)
      sothat(user knows what to do next)
  when(unknown subcommand provided)
    then(shows error: "unknown subcommand: X")
    then(shows available subcommands)
    then(exits with non-zero code)
      sothat(user can correct their invocation)

## usecase.4 = preserve subcommand args

given(a user passes args to a subcommand)
  when(invokes "git.branch.rebase begin --mode apply")
    then(passes "--mode apply" to git.branch.rebase.begin)
      sothat(subcommand receives its options)
  when(invokes "git.branch.rebase begin --mode plan")
    then(passes "--mode plan" to git.branch.rebase.begin)

## usecase.5 = backward compat via aliases

given(a user invokes old-style skill names)
  when(invokes "git.branch.rebase.begin")
    then(skill still works via rhachet lookup)
      sothat(extant invocations are not broken)
  when(invokes "git.branch.rebase.continue")
    then(skill still works via rhachet lookup)
  when(invokes "git.branch.rebase.abort")
    then(skill still works via rhachet lookup)
