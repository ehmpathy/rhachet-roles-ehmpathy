# blackbox criteria: git.repo.test snapshots in sync

## usecase.1 = run tests locally

given([case1] mechanic runs integration tests)
  when([t0] tests execute with default flags)
    then(all snapshot assertions pass)
      sothat(mechanic has confidence output matches contract)
    then(exit code is 0)
      sothat(mechanic knows tests passed)

given([case2] mechanic runs integration tests after code change)
  when([t0] code change affects output format)
    then(snapshot assertion fails with diff)
      sothat(mechanic sees exactly what output differs)
    then(exit code is non-zero)
      sothat(mechanic knows to investigate)

## usecase.2 = run tests on CI

given([case1] CI runs test workflow)
  when([t0] workflow executes with THOROUGH=true)
    then(all 85 snapshot tests execute)
      sothat(no snapshot is skipped due to changed-only filter)
    then(hand-edited snapshots that do not match reality cause failure)
      sothat(lies are caught immediately)

given([case2] prior contributor hand-edited a snapshot)
  when([t0] CI runs full suite)
    then(test fails because fixture output does not match snapshot)
      sothat(the lie is exposed)
    then(error shows which snapshot mismatches)
      sothat(contributor can diagnose the issue)

## usecase.3 = update snapshots intentionally

given([case1] mechanic changes output format deliberately)
  when([t0] tests run with --resnap flag)
    then(snapshots are updated to match new output)
      sothat(new behavior becomes the contract)
  when([t1] mechanic commits updated snapshots)
    then(PR diff shows snapshot changes)
      sothat(reviewers can verify the change was intentional)

## usecase.4 = diagnose fixture issues

given([case1] snapshot test fails)
  when([t0] mechanic reviews the diff)
    then(diff shows expected vs actual output)
      sothat(mechanic can identify what fixture produces wrong output)
    then(diff is readable after sanitization of timestamps, paths, versions)
      sothat(mechanic focuses on meaningful differences)

given([case2] mechanic suspects snapshot was hand-edited incorrectly)
  when([t0] mechanic runs git blame on snapshot file)
    then(commit history shows who edited and when)
      sothat(mechanic can determine if edit was intentional or mistake)

## usecase.5 = preview test scope

given([case1] mechanic wants to see what tests will run)
  when([t0] tests run in plan mode without --thorough)
    then(only files changed since main are listed)
      sothat(mechanic gets fast feedback in development)
  when([t1] tests run in plan mode with --thorough)
    then(all test files in scope are listed)
      sothat(mechanic can verify full coverage)
