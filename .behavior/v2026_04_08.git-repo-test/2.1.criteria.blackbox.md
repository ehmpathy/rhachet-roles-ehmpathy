# criteria.blackbox: git.repo.test

blackbox criteria for the git.repo.test skill extension.

---

# usecase.1 = run unit tests

given(repo with test:unit command)
  when(--what unit)
    then(runs npm run test:unit)
    then(captures stdout to log file)
    then(captures stderr to log file)
    then(shows summary with status, stats, log paths)
      sothat(clone sees result without raw jest output)
    then(exit 0 on pass)
    then(exit 2 on fail)

  when(--what unit --scope pattern)
    then(passes --testPathPattern to jest)
    then(filters tests to match pattern)
      sothat(clone runs only relevant tests)

  when(--what unit --resnap)
    then(sets RESNAP=true env var)
    then(snapshots update)
      sothat(clone updates snapshots in one command)

# usecase.2 = run integration tests

given(repo with test:integration command)
  when(--what integration)
    then(unlocks keyrack ehmpath/test first)
      sothat(credentials available for external services)
    then(runs npm run test:integration)
    then(shows keyrack unlock in output)
    then(shows summary with status, stats, log paths)
    then(exit 0 on pass)
    then(exit 2 on fail)

  when(--what integration --scope pattern)
    then(unlocks keyrack first)
    then(passes --testPathPattern to jest)

  when(--what integration --resnap)
    then(unlocks keyrack first)
    then(sets RESNAP=true env var)

# usecase.4 = run lint (extant behavior)

given(repo with test:lint command)
  when(--what lint)
    then(runs npm run test:lint)
    then(does NOT unlock keyrack)
      sothat(no unnecessary credential access)
    then(shows summary with status, defect count, log paths)
    then(exit 0 on pass)
    then(exit 2 on fail)

  when(--what lint --resnap)
    then(ignores --resnap flag)
      sothat(lint has no snapshots to update)

  when(--what lint --scope pattern)
    then(ignores --scope flag)
      sothat(lint runs on all files)

---

# usecase.5 = pass raw args to jest

given(repo with test:unit command)
  when(--what unit -- --testNamePattern="pattern")
    then(passes args after -- to npm)
    then(jest receives --testNamePattern)
      sothat(clone filters by test name)

  when(--what unit --scope path -- --verbose)
    then(combines --scope with raw args)
    then(jest receives both --testPathPattern and --verbose)

---

# usecase.6 = fail fast on no tests matched

given(repo with test:unit command)
  when(--what unit --scope nonExistentPattern)
    then(detects no tests match)
    then(exit 2 with constraint error)
    then(shows helpful hint about scope pattern)
      sothat(clone knows to check scope, not debug jest)

---

# usecase.7 = fail fast on absent command

given(repo without test:unit command)
  when(--what unit)
    then(exit 2 with constraint error)
    then(shows error: no test:unit command)
    then(shows hint about ehmpathy convention)
      sothat(clone learns expected npm commands)

given(repo without test:integration command)
  when(--what integration)
    then(exit 2 with constraint error)
    then(shows error: no test:integration command)

# usecase.8 = output format (progressive)

given(any test run)
  when(test begins)
    then(shows "lets ride..." header immediately)
    then(shows skill name and args)
    then(shows keyrack status if integration/acceptance)
    then(shows scope preview if --scope used)
      then(shows scope pattern)
      then(shows matched file count)
      sothat(clone can verify scope matched intended files)
    then(starts timer that prints "status: inflight (Ns)" each second)
      sothat(clone knows skill is active)
      note(timer lines stack, not overwrite)

  when(tests complete)
    then(stops timer)
    then(shows status: passed | failed)
    then(shows stats nested)
      then(shows suites: N files)
      then(shows tests: X passed, Y failed, Z skipped)
      then(shows time: X.Xs)
    then(shows log nested)
      then(shows stdout path)
      then(shows stderr path)
    then(shows tip on failure)

# usecase.9 = log capture

given(any test run)
  when(tests complete successfully)
    then(stdout captured to .log/.../timestamp.stdout.log)
    then(stderr captured to .log/.../timestamp.stderr.log)
    then(log paths shown in output)

  when(tests fail)
    then(stdout captured to .log/.../timestamp.stdout.log)
    then(stderr captured to .log/.../timestamp.stderr.log)
    then(log paths shown in output)
      sothat(clone can read full details)

---

# usecase.10 = keyrack unlock behavior

given(keyrack locked)
  when(--what integration)
    then(unlocks ehmpath/test before tests)
    then(shows keyrack status in output)

given(keyrack already unlocked)
  when(--what integration)
    then(unlock is idempotent, no error)
    then(shows keyrack status in output)

given(keyrack unlock fails)
  when(--what integration)
    then(exit 1 with malfunction error)
    then(shows keyrack error in output)
      sothat(clone knows to check keyrack setup)

---

# usecase.11 = context efficiency

given(any test run)
  when(tests complete)
    then(output is summary only)
    then(no raw jest output in terminal)
    then(full output only in log files)
      sothat(clone context window stays clean)

# usecase.12 = run all tests

given(repo with all test commands)
  when(--what all)
    then(runs lint first)
    then(runs unit second)
    then(runs integration third)
    then(runs acceptance fourth)
    then(unlocks keyrack before integration)
    then(emits status block as each type completes)
      sothat(clone sees progress)
    then(shows total duration across all types)
    then(shows log paths for each type)
    then(exit 0 if all pass)
    then(exit 2 on first failure)

  when(--what all and lint fails)
    then(stops after lint)
    then(does not run unit, integration, acceptance)
      sothat(clone addresses failure before next type runs)
    then(exit 2)

  when(--what all and unit fails)
    then(stops after unit)
    then(does not run integration, acceptance)
    then(exit 2)

# usecase.13 = thorough mode

given(repo with test commands)
  when(--what unit --thorough)
    then(sets THOROUGH=true env var)
    then(runs full test suite regardless of impact analysis)
      sothat(clone verifies all tests pass)

  when(--what all --thorough)
    then(sets THOROUGH=true for each test type)
    then(runs full suite for lint, unit, integration, acceptance)

---

# usecase.14 = namespaced log paths

given(any test run)
  when(--what lint)
    then(logs to .log/.../what=lint/timestamp.stdout.log)
    then(logs to .log/.../what=lint/timestamp.stderr.log)

  when(--what unit)
    then(logs to .log/.../what=unit/timestamp.stdout.log)
    then(logs to .log/.../what=unit/timestamp.stderr.log)

  when(--what integration)
    then(logs to .log/.../what=integration/timestamp.stdout.log)
    then(logs to .log/.../what=integration/timestamp.stderr.log)

  when(--what acceptance)
    then(logs to .log/.../what=acceptance/timestamp.stdout.log)
    then(logs to .log/.../what=acceptance/timestamp.stderr.log)

  when(--what all)
    then(each type logs to its own namespaced directory)
      sothat(logs do not overlap between test types)

# usecase.15 = scope preview

given(repo with test:unit command)
  when(--what unit --scope pattern)
    then(shows scope pattern before test runs)
    then(shows matched file count)
      sothat(clone can verify intended files matched)
    then(shows common directory patterns if detectable)
      sothat(clone can verify scope is not too broad)

  when(--what unit --scope matches zero files)
    then(shows matched: 0 files before exit)
    then(exit 2 with constraint error)
      sothat(clone knows scope matched no files)
