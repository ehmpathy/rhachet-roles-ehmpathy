### [REUSE] runInTempGitRepo helper pattern

```typescript
const runInTempGitRepo = (args: {
  packageJson?: object | null;
  eslintConfig?: object | null;
  sourceFiles?: Record<string, string>;
  testLintScript?: string;
  gitRepoTestArgs: string[];
}): { stdout: string; stderr: string; exitCode: number; tempDir: string } => {
  const tempDir = genTempDir({ slug: 'git-repo-test', git: true });

  // create package.json
  if (args.packageJson !== null) {
    const pkgJson = args.packageJson ?? { ... };
    fs.writeFileSync(path.join(tempDir, 'package.json'), JSON.stringify(pkgJson, null, 2));
  }

  // create source files
  if (args.sourceFiles) {
    for (const [filePath, content] of Object.entries(args.sourceFiles)) {
      const fullPath = path.join(tempDir, filePath);
      fs.mkdirSync(path.dirname(fullPath), { recursive: true });
      fs.writeFileSync(fullPath, content);
    }
  }

  // run git.repo.test.sh
  const result = spawnSync('bash', [scriptPath, ...args.gitRepoTestArgs], {
    cwd: tempDir,
    encode: 'utf-8',
    stdio: ['pipe', 'pipe', 'pipe'],
  });

  return {
    stdout: result.stdout ?? '',
    stderr: result.stderr ?? '',
    exitCode: result.status ?? 1,
    tempDir,
  };
};
```

## 3. test fixture patterns

### [NEW] jest fixture for unit tests

```typescript
const setupJestUnitFixture = (tempDir: string, args: {
  testFiles?: Record<string, string>;
  pass?: boolean;
}) => {
  // create jest.config.js
  fs.writeFileSync(
    path.join(tempDir, 'jest.config.js'),
    `module.exports = { testEnvironment: 'node' };`,
  );

  // create package.json with test:unit
  const pkgJson = {
    name: 'test-repo',
    scripts: {
      'test:unit': 'jest --testPathPattern=\\.test\\.ts$',
    },
    devDependencies: {
      jest: '^29.0.0',
    },
  };
  fs.writeFileSync(path.join(tempDir, 'package.json'), JSON.stringify(pkgJson, null, 2));

  // create test files
  if (args.testFiles) {
    for (const [filePath, content] of Object.entries(args.testFiles)) {
      const fullPath = path.join(tempDir, filePath);
      fs.mkdirSync(path.dirname(fullPath), { recursive: true });
      fs.writeFileSync(fullPath, content);
    }
  } else {
    // default pass/fail test
    const testContent = args.pass
      ? `test('passes', () => { expect(1).toBe(1); });`
      : `test('fails', () => { expect(1).toBe(2); });`;
    fs.mkdirSync(path.join(tempDir, 'src'), { recursive: true });
    fs.writeFileSync(path.join(tempDir, 'src/example.test.ts'), testContent);
  }
};
```

### [NEW] mock npm test command

const setupMockTestCommand = (tempDir: string, args: {
  what: 'unit' | 'integration' | 'acceptance';
  stdout: string;
  stderr?: string;
  exitCode: number;
}) => {
  const pkgJson = {
    name: 'test-repo',
    scripts: {
      [`test:${args.what}`]: `echo "${args.stdout}" && ${args.stderr ? `echo "${args.stderr}" >&2 &&` : ''} exit ${args.exitCode}`,
    },
  };
  fs.writeFileSync(path.join(tempDir, 'package.json'), JSON.stringify(pkgJson, null, 2));
};
