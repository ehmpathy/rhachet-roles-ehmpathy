  then('accepts: --thorough (optional)')

## composition boundaries

given('test execution flow')
  then('validates --what argument first')
  then('validates npm command exists second')
  then('unlocks keyrack if integration|acceptance')
  then('constructs npm command with flags')
  then('executes and captures to temp files')
  then('parses output for stats')
  then('detects no-tests-matched condition')
  then('emits summary to terminal')
  then('persists logs to .log/')
  then('returns appropriate exit code')

given('flag composition')
  then('--scope becomes --testPathPattern for jest')
  then('--resnap becomes RESNAP=true env var')
  then('--thorough becomes THOROUGH=true env var')
  then('-- args pass through to npm verbatim')
  then('lint ignores --scope and --resnap')
  then('--what all runs lint → unit → integration → acceptance in sequence')

given('log management')
  then('creates .log/role=mechanic/skill=git.repo.test/what=${TYPE}/ if absent')
  then('namespaces logs by test type to prevent overlap')
  then('findserts .gitignore in log directory')
  then('uses ISO timestamp for log filenames')
  then('captures stdout and stderr separately')
