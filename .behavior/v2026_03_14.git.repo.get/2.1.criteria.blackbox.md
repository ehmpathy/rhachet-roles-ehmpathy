# blackbox criteria: git.repo.get

## usecase.1 = list repos

```
given(mechanic wants to discover repos)
  when(repos --repos 'ehmpathy/*')
    then(lists all ehmpathy repos)
      sothat(mechanic knows what repos exist)
    then(shows local path for local repos)
    then(shows github url for cloud repos)
    then(header says "far out" on success)

  when(repos --repos 'sql-*')
    then(lists sql-* repos in any org)
      sothat(mechanic can search by repo name pattern)

  when(repos --repos 'ehmpathy/*-roles-*')
    then(lists repos that match *-roles-* in ehmpathy)

  when(repos --repos '@local/ehmpathy/*')
    then(lists only local clones)
      sothat(mechanic can scope to local)

  when(repos --repos '@cloud/ehmpathy/*')
    then(lists only cloud repos)
      sothat(mechanic can scope to cloud)

  when(repos --repos 'nonexistent-org/*')
    then(header says "crickets...")
    then(shows found: 0 repos)

  when(repos)
    then(defaults to --repos '*/*')
    then(lists all local repos)
      sothat(mechanic can discover all repos without --repos flag)
```

## usecase.2 = list files

```
given(mechanic wants to browse files in a repo)
  when(files --in ehmpathy/domain-objects)
    then(lists file tree for repo)
    then(reads from origin/main by default)
    then(header says "far out" on success)

  when(files --in ehmpathy/domain-objects --paths '**/*.test.ts')
    then(lists only files that match glob)
      sothat(mechanic can filter by name)

  when(files --in ehmpathy/domain-objects --words 'DomainEntity')
    then(lists only files with pattern in content)
      sothat(mechanic can find files by content)

  when(files --in ehmpathy/domain-objects --paths 'src/**/*.ts' --words 'export')
    then(lists ts files in src/ with 'export' in content)
      sothat(mechanic can combine filters)

  when(files --in ehmpathy/domain-objects --ref 'v1.0.0')
    then(lists files at that tag)
      sothat(mechanic can browse specific versions)

  when(files --in ehmpathy/unknown-repo)
    then(header says "bummer dude")
    then(shows error: repo not found)

given(mechanic wants to find files across repos)
  when(files --repos 'ehmpathy/*' --words 'sdkQuotes')
    then(searches across all ehmpathy repos)
    then(lists files with 'sdkQuotes' in content)
    then(groups results by repo)
      sothat(mechanic can find which repos use a pattern)

  when(files --words 'sdkQuotes')
    then(defaults to --repos '*/*')
    then(searches across all local repos)
      sothat(mechanic can search everywhere without --repos flag)

  when(files --repos 'ehmpathy/*' --paths '**/*.ts')
    then(lists .ts files across all ehmpathy repos)
      sothat(mechanic can filter by path across repos)
```

## usecase.3 = search lines

```
given(mechanic wants to search for a pattern)
  when(lines --in ehmpathy/domain-objects --words 'DomainEntity')
    then(shows matches with 21 lines radius by default)
      sothat(mechanic sees context around each match)
    then(marks match lines with > prefix)
    then(groups results by file)
    then(header says "far out" on success)

  when(lines --in ehmpathy/domain-objects --words 'DomainEntity' --radius 5)
    then(shows matches with 5 lines radius)
      sothat(mechanic can control context size)

  when(lines --repos 'ehmpathy/*' --words 'DomainEntity')
    then(searches across all ehmpathy repos)
    then(groups results by repo then by file)
      sothat(mechanic can search cross-repo)

  when(lines --in ehmpathy/domain-objects --words 'NonExistentPattern')
    then(header says "crickets...")
    then(shows found: 0 matches)

  when(lines --in ehmpathy/domain-objects --paths 'src/**/*.ts' --words 'export')
    then(searches only in files that match path glob)
      sothat(mechanic can narrow search scope)

given(mechanic wants to search across all repos)
  when(lines --repos 'ehmpathy/*' --words 'sdkQuotes')
    then(searches across all ehmpathy repos)
    then(groups results by repo then by file)
    then(shows match count per repo)
      sothat(mechanic can find which repos use a pattern)

  when(lines --words 'sdkQuotes')
    then(defaults to --repos '*/*')
    then(searches across all local repos)
      sothat(mechanic can search everywhere without --repos flag)

  when(lines --repos '*/*' --words 'DomainEntity' --paths '**/*.ts')
    then(searches .ts files across all repos)
      sothat(mechanic can combine multi-repo with path filter)
```

```
    then(header says "crickets...")
    then(shows found: 0 files)
```

## usecase.5 = source scope behavior

```
given(mechanic has local clone at ~/git/ehmpathy/domain-objects)
  when(lines --in ehmpathy/domain-objects --words 'pattern')
    then(reads from local clone)
      sothat(local-first is default)
    then(shows (local) indicator)

given(mechanic does not have local clone)
  when(lines --in ehmpathy/some-repo --words 'pattern')
    then(falls back to cloud via gh cli)
    then(shows (cloud) indicator)
      sothat(cloud fallback works transparently)

given(mechanic wants explicit scope)
  when(lines --repos '@local/ehmpathy/*' --words 'pattern')
    then(only searches local repos)
    then(does not fall back to cloud)

  when(lines --repos '@cloud/ehmpathy/*' --words 'pattern')
    then(only searches cloud repos)
    then(does not use local clones)
```
