## pattern.1 = temp git repo fixture

**[EXTEND]** — add global storage directory support

### citations

[1] `git.commit.uses.integration.test.ts:13-41`
```typescript
const runInTempGitRepo = (args: {
  args: string[];
  meterState?: { uses: number; push: string };
}): { stdout: string; stderr: string; exitCode: number; tempDir: string } => {
  const tempDir = genTempDir({ slug: 'git-commit-uses-test', git: true });

  // create .meter directory and state if provided
  if (args.meterState) {
    const meterDir = path.join(tempDir, '.meter');
    fs.mkdirSync(meterDir, { recursive: true });
    fs.writeFileSync(
      path.join(meterDir, 'git.commit.uses.jsonc'),
      JSON.stringify(args.meterState, null, 2),
    );
  }

  const result = spawnSync('bash', [skillPath, ...args.args], {
    cwd: tempDir,
    encoding: 'utf-8',
    stdio: ['pipe', 'pipe', 'pipe'],
  });

  return {
    stdout: result.stdout ?? '',
    stderr: result.stderr ?? '',
    exitCode: result.status ?? 1,
    tempDir,
  };
};
```

### relation to wish

extend to support:
- `globalBlockerState?: { blocked: boolean }` — creates `~/.rhachet/storage/.../git.commit.uses.jsonc`
- tests will need to use a temp `$HOME` or mock the global storage path

given('[case1] set --quant N --push block', () => {
  when('[t0] setting 3 uses with push blocked', () => {

then('state file is created', () => {

    args: ['set', '--quant', '3', '--push', 'block'],
