## reproduction feasibility

### test utilities available

from git.commit skills research:
- `genTempDir({ git: true })` — creates temp git repo
- spawnSync pattern — executes skill
- meter state fixture — sets permission
- branch creation — feature branch setup
- commit creation — commits to rebase

### setup required per scenario

| scenario | setup |
|----------|-------|
| plan preview | feature branch with commits, main behind |
| successful rebase | feature branch, main has commits, no conflicts |
| conflict encounter | feature branch modifies same lines as main |
| conflict continue | rebase in progress, conflicts settled and staged |
| abort rebase | rebase in progress |
| no permission | meter file with push: 'block' |
| on main branch | stay on main branch |
| dirty work tree | unstaged changes |
| already up to date | feature branch, no commits ahead |

### concrete test sketch

```typescript
import { spawnSync } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import { genTempDir, given, then, when } from 'test-fns';

describe('git.branch.rebase.begin.sh', () => {
  const skillPath = path.join(__dirname, 'git.branch.rebase.begin.sh');

  const setupRebaseScenario = (args: {
    meterState?: { uses: number; push: string };
    commitsOnFeature?: string[];
    commitsOnMainAfter?: string[];
    conflictFile?: { featureContent: string; mainContent: string };
  }): string => {
    const tempDir = genTempDir({
      slug: 'git-rebase-test',
      git: true,
      symlink: [{ at: 'node_modules', to: 'node_modules' }],
    });

    // setup user
    spawnSync('git', ['config', 'user.name', 'Test Human'], { cwd: tempDir });
    spawnSync('git', ['config', 'user.email', 'human@test.com'], { cwd: tempDir });

    // setup meter
    if (args.meterState) {
      const meterDir = path.join(tempDir, '.meter');
      fs.mkdirSync(meterDir, { recursive: true });
      fs.writeFileSync(
        path.join(meterDir, 'git.commit.uses.jsonc'),
        JSON.stringify(args.meterState, null, 2),
      );
      fs.writeFileSync(path.join(tempDir, '.gitignore'), '.meter/\n');
      spawnSync('git', ['add', '.gitignore'], { cwd: tempDir });
      spawnSync('git', ['commit', '-m', 'setup'], { cwd: tempDir });
    }

    // create feature branch
    spawnSync('git', ['checkout', '-b', 'fix/feature'], { cwd: tempDir });

    // add commits on feature
    if (args.commitsOnFeature) {
      for (const [i, msg] of args.commitsOnFeature.entries()) {
        fs.writeFileSync(path.join(tempDir, `feature-${i}.txt`), `content ${i}`);
        spawnSync('git', ['add', '-A'], { cwd: tempDir });
        spawnSync('git', ['commit', '-m', msg], { cwd: tempDir });
      }
    }

    // go back to main, add commits, go back to feature
    if (args.commitsOnMainAfter) {
      spawnSync('git', ['checkout', 'main'], { cwd: tempDir });
      for (const [i, msg] of args.commitsOnMainAfter.entries()) {
        fs.writeFileSync(path.join(tempDir, `main-${i}.txt`), `main content ${i}`);
        spawnSync('git', ['add', '-A'], { cwd: tempDir });
        spawnSync('git', ['commit', '-m', msg], { cwd: tempDir });
      }
      spawnSync('git', ['checkout', 'fix/feature'], { cwd: tempDir });
    }

    return tempDir;
  };

  given('[case1] successful rebase scenario', () => {
    when('[t0] mechanic runs plan mode', () => {
      then('shows preview', () => {
        const tempDir = setupRebaseScenario({
          meterState: { uses: 3, push: 'allow' },
          commitsOnFeature: ['fix: change 1', 'fix: change 2'],
          commitsOnMainAfter: ['chore: main update'],
        });

        // note: 'utf-8' is library API for spawnSync
        const result = spawnSync('bash', [skillPath], {
          cwd: tempDir,
          encoding: 'utf-8',
        });

        expect(result.status).toBe(0);
        expect(result.stdout).toContain('heres the wave');
        expect(result.stdout).toContain('commits: 2');
        expect(result.stdout).toContain('behind: 1');
      });
    });
  });
});
```
